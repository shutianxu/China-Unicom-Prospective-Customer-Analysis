#用户游览情况-count字段待解决
drop table temp_autohome_device_iu;
create table temp_autohome_device_iu as
select count(1),device_number,step,shijian from 
(select distinct device_number,step,substring(start_time,1,10) 
shijianfrom temp_autohome_step_iu
)m group by device_number,step,shijian;


drop table if exists temp_autohome_device_ia;
create table temp_autohome_device_ia as
select 
distinct t1.device_number,
t1.step,substring(t1.start_time,1,10) shijian,
t2.cishu from temp_autohome_step_iu t1,
(select t.device_number,count(t.device_number) cishu from temp_autohome_step_iu t group by device_number) t2
where t1.device_number = t2.device_number;



#用户游览情况-count字段解决
drop table temp_autohome_device_iu;
create table temp_autohome_device_iu as
select device_number,substring(start_time,1,10) as shijian,step,count(*) as men
from temp_autohome_step_iu group by device_number,substring(start_time,1,10),step
